 ===== GEOCODE =====
async function geocode(address) {
  const res = await fetch(
    httpsnominatim.openstreetmap.orgsearchformat=json&q= + encodeURIComponent(address)
  );
  const data = await res.json();
  if (data.length === 0) return null;
  return [parseFloat(data[0].lat), parseFloat(data[0].lon)];
}

 ===== ICON =====

 ===== DISTANCE (simple) =====
function distance(a, b) {
  return Math.sqrt(
    (a[0]-b[0])2 + (a[1]-b[1])2
  );
}

 ===== GREEDY OPTIMIZE =====
function optimizeRoute(points, addresses) {

  let orderedPoints = [points[0]];
  let orderedAddresses = [addresses[0]];

  let remainingPoints = points.slice(1);
  let remainingAddr = addresses.slice(1);

  while (remainingPoints.length  0) {
    let last = orderedPoints[orderedPoints.length - 1];

    let minIndex = 0;
    let minDist = distance(last, remainingPoints[0]);

    for (let i = 1; i  remainingPoints.length; i++) {
      let d = distance(last, remainingPoints[i]);
      if (d  minDist) {
        minDist = d;
        minIndex = i;
      }
    }

    orderedPoints.push(remainingPoints[minIndex]);
    orderedAddresses.push(remainingAddr[minIndex]);

    remainingPoints.splice(minIndex, 1);
    remainingAddr.splice(minIndex, 1);
  }

  return { orderedPoints, orderedAddresses };
}

 ===== OSRM ROUTE =====
async function drawRoute(points) {

  const coords = points.map(p = p[1]+,+p[0]).join(;);

  const url = `httpsrouter.project-osrm.orgroutev1driving${coords}overview=full&geometries=geojson`;

  const res = await fetch(url);
  const data = await res.json();

  const route = data.routes[0];

  if (routeLine) map.removeLayer(routeLine);

  routeLine = L.geoJSON(route.geometry, {
    style { color 'blue', weight 5 }
  }).addTo(map);

  map.fitBounds(routeLine.getBounds());

  document.getElementById(info).innerText =
    Tổng quãng đường  + (route.distance1000).toFixed(2) +  km;
}

 ===== BUILD =====
async function buildRoute() {

  const inputs = document.querySelectorAll(.point-input);

  let addresses = [];
  inputs.forEach(i = {
    if (i.value) addresses.push(i.value);
  });

  if (addresses.length  2) {
    alert(Nhập ít nhất 2 điểm);
    return;
  }

  markers.forEach(m = map.removeLayer(m));
  markers = [];
  if (routeLine) map.removeLayer(routeLine);

  let points = [];

  for (let addr of addresses) {
    const p = await geocode(addr);
    if (!p) {
      alert(Không tìm thấy  + addr);
      return;
    }
    points.push(p);
  }

   ===== OPTIMIZE =====
  const { orderedPoints, orderedAddresses } =
    optimizeRoute(points, addresses);

   ===== GỢI Ý CUNG ĐƯỜNG =====
  let html = ;
  orderedAddresses.forEach((addr, i) = {
    html += `b${i+1}b. ${addr}br`;
  });

  document.getElementById(suggestion).innerHTML = html;

   ===== MARKER =====
  orderedPoints.forEach((p, i) = {
    const marker = L.marker(p, {
      icon createNumberIcon(i+1)
    })
    .addTo(map)
    .bindPopup(`bĐiểm ${i+1}bbr${orderedAddresses[i]}`);

    markers.push(marker);
  });

   ===== ROUTE =====
  await drawRoute(orderedPoints);
}
